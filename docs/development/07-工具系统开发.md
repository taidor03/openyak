# OpenYak 工具系统开发文档

## 概述

工具（Tool）是 Agent 执行具体操作的接口。每个工具继承 `ToolDefinition` 基类，实现 `execute()` 方法，定义 JSON Schema 参数，返回 `ToolResult`。OpenYak 内置 24 个工具，覆盖文件操作、搜索、执行、交互、网络等场景。

## 工具架构

### ToolDefinition 基类

**位置**：`backend/app/tool/base.py`

```python
class ToolDefinition(ABC):
    @property
    @abstractmethod
    def id(self) -> str:
        """工具唯一标识符"""

    @property
    @abstractmethod
    def description(self) -> str:
        """工具描述（发送给 LLM）"""

    @property
    def is_concurrency_safe(self) -> bool:
        """是否可并发执行（只读工具返回 True）"""
        return False

    @abstractmethod
    def parameters_schema(self) -> dict:
        """返回 JSON Schema 参数定义"""

    @abstractmethod
    async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
        """执行工具逻辑"""
```

### ToolResult

```python
@dataclass
class ToolResult:
    output: str = ""                          # 输出文本
    metadata: dict[str, Any] = field(default_factory=dict)  # 元数据
    title: str | None = None                  # 标题
    error: str | None = None                  # 错误信息
    attachments: list[dict] = field(default_factory=list)  # 文件附件

    @property
    def success(self) -> bool:
        return self.error is None
```

### ToolContext

```python
class ToolContext:
    session_id: str           # 当前会话 ID
    message_id: str           # 当前消息 ID
    agent: str                # 当前 agent 名称
    abort: Callable[[], None] # 中止执行的回调
    metadata: dict[str, Any]  # 额外元数据
```

### ToolRegistry

**位置**：`backend/app/tool/registry.py`

```python
class ToolRegistry:
    def register(self, tool: ToolDefinition) -> None
    def get(self, tool_id: str) -> ToolDefinition | None
    def list_tools(self) -> list[ToolDefinition]
    def filter_by_agent(self, agent: AgentInfo) -> list[ToolDefinition]
```

## 创建自定义工具

### 步骤 1：创建工具类

在 `backend/app/tool/builtin/` 创建新文件 `my_tool.py`：

```python
"""My custom tool."""

import logging
from app.tool.base import ToolDefinition, ToolResult
from app.tool.context import ToolContext
from app.tool.truncation import truncate_output

logger = logging.getLogger(__name__)


class MyTool(ToolDefinition):
    @property
    def id(self) -> str:
        return "my_tool"

    @property
    def description(self) -> str:
        return "My custom tool that does something useful"

    @property
    def is_concurrency_safe(self) -> bool:
        return True  # 只读工具可并发

    def parameters_schema(self) -> dict:
        return {
            "type": "object",
            "properties": {
                "param1": {
                    "type": "string",
                    "description": "First parameter",
                },
                "param2": {
                    "type": "integer",
                    "description": "Second parameter (optional)",
                    "default": 0,
                },
            },
            "required": ["param1"],
        }

    async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
        try:
            param1 = args["param1"]
            param2 = args.get("param2", 0)

            # 执行工具逻辑
            result = self._process(param1, param2)

            # 截断大输出（重要！）
            output = truncate_output(result)

            return ToolResult(
                output=output,
                metadata={"param1": param1, "param2": param2},
            )
        except Exception as e:
            logger.error(f"MyTool execution failed: {e}", exc_info=True)
            return ToolResult(error=str(e))

    def _process(self, param1: str, param2: int) -> str:
        """实际处理逻辑"""
        return f"Processed '{param1}' with value {param2}"
```

### 步骤 2：注册工具

在 `backend/app/tool/builtin/__init__.py` 中注册：

```python
from .my_tool import MyTool

def register_builtin_tools(registry: ToolRegistry) -> None:
```

### 步骤 4：测试工具

```python
# tests/test_tool/test_my_tool.py
import pytest
from app.tool.builtin.my_tool import MyTool
from app.tool.context import ToolContext

@pytest.mark.asyncio
async def test_my_tool_basic():
    tool = MyTool()
    assert tool.id == "my_tool"
    assert tool.is_concurrency_safe is True

    ctx = ToolContext(
        session_id="test",
        message_id="test",
        agent="build",
        abort=lambda: None,
        metadata={}
    )

    result = await tool.execute({"param1": "hello", "param2": 42}, ctx)
    assert result.success
    assert "hello" in result.output

@pytest.mark.asyncio
async def test_my_tool_missing_required():
    tool = MyTool()
    ctx = ToolContext(session_id="t", message_id="t", agent="build",
                      abort=lambda: None, metadata={})

    result = await tool.execute({}, ctx)
    assert result.error is not None
```

## 工具执行流程

```
LLM 生成 tool_call
  ↓
SessionProcessor 解析 tool_calls
  ↓
ToolExecutor.execute()
  ├── 参数验证（JSON Schema）
  ├── 权限检查（evaluate()）
  │   ├── allow → 继续执行
  │   ├── ask → 发送 permission_request SSE
  │   │        → 等待用户响应
  │   │        → allow → 继续执行
  │   │        → deny → 返回错误
  │   └── deny → 返回错误
  ├── doom loop 检测
  ├── ToolDefinition.execute(args, ctx)
  ├── 输出截断（truncate_output）
  ├── 保存 ToolPart 到数据库
  └── 发送 SSE 事件（tool_result / tool_error）
```

## 工具最佳实践

### 输出截断

**必须**对大输出使用截断，避免 token 溢出：

```python
from app.tool.truncation import truncate_output

async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
    output = self._get_large_output()
    truncated = truncate_output(output, max_chars=30000)
    return ToolResult(output=truncated)
```

### 并发安全标记

- **只读工具**：标记为 `is_concurrency_safe = True`，可被并发调用
- **写入工具**：标记为 `is_concurrency_safe = False`，串行执行

### 错误处理

```python
async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
    try:
        result = await self._do_work(args)
        return ToolResult(output=result)
    except FileNotFoundError as e:
        return ToolResult(error=f"文件不存在: {e}")
    except PermissionError as e:
        return ToolResult(error=f"权限不足: {e}")
    except Exception as e:
        logger.error(f"Unexpected error in {self.id}", exc_info=True)
        return ToolResult(error=f"执行失败: {e}")
```

### 中止支持

长时间运行的工具应检查中止信号：

```python
async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
    for item in long_list:
        # 检查中止
        if ctx.metadata.get("aborted"):
            return ToolResult(error="用户已中止")
        # 处理 item
        await process(item)
    return ToolResult(output="完成")
```

### 文件附件

返回文件附件供前端展示：

```python
return ToolResult(
    output="报告已生成",
    attachments=[
        {
            "type": "file",
            "path": "/path/to/report.pdf",
            "mime_type": "application/pdf",
            "name": "report.pdf"
        }
    ]
)
```

### 使用上下文信息

```python
async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
    session_id = ctx.session_id
    agent_name = ctx.agent
    # 基于上下文执行不同逻辑
    return ToolResult(output=f"Session: {session_id}")
```

## 常见问题

### 如何访问数据库？

使用 session_factory 创建数据库会话：

```python
from sqlalchemy.ext.asyncio import AsyncSession

async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
    async with self.session_factory() as db:
        result = await db.execute(select(MyModel))
        # ...
```

### 如何处理超时？

```python
import asyncio

async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
    try:
        result = await asyncio.wait_for(
            self._long_operation(),
            timeout=30.0
        )
        return ToolResult(output=str(result))
    except asyncio.TimeoutError:
        return ToolResult(error="操作超时（30秒）")
```

```toml
]
```
