# examples/tools/sql_query_tool.py
"""SQL query tool for database operations."""

import asyncio
from typing import Any
from app.tool.base import ToolDefinition, ToolResult, ToolContext


class SQLQueryTool(ToolDefinition):
    """Execute SQL queries on the application database."""
    
    def __init__(self, db_url: str):
        self.db_url = db_url
        self._engine = None
    
    @property
    def id(self) -> str:
        return "sql_query"
    
    @property
    def description(self) -> str:
        """
        Execute SQL queries on the application database.
        
        ⚠️ 警告：只允许 SELECT 查询，禁止修改操作！
        """
        return "执行 SQL 查询（仅 SELECT，禁止修改）"
    
    @property
    def is_concurrency_safe(self) -> bool:
        return True
    
    def parameters_schema(self) -> dict:
        return {
            "type": "object",
            "properties": {
                "query": {
                    "type": "string",
                    "description": "SQL 查询语句（仅 SELECT）",
                },
                "params": {
                    "type": "object",
                    "description": "查询参数",
                },
            },
            "required": ["query"],
        }
    
    async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
        """Execute SQL query."""
        query = args["query"]
        params = args.get("params", {})
        
        # 安全检查：只允许 SELECT
        if not self._is_safe_query(query):
            return ToolResult(
                error="⚠️ 禁止执行非 SELECT 查询！"
            )
        
        try:
            from sqlalchemy import text
            from app.db import engine
            
            async with engine.begin() as conn:
                result = await conn.execute(
                    text(query),
                    params
                )
                
                # 处理结果
                rows = [dict(row._mapping) for row in result.fetchall()]
                
                output_lines = [
                    f"查询成功！返回 {len(rows)} 行数据：",
                    "",
                ]
                
                if rows:
                    import json
                    output_lines.append(json.dumps(rows, indent=2, ensure_ascii=False))
                else:
                    output_lines.append("未返回数据。")
                
                return ToolResult(
                    output="\n".join(output_lines),
                    metadata={"row_count": len(rows)},
                )
        except Exception as e:
            return ToolResult(error=f"查询失败：{str(e)}")
    
    def _is_safe_query(self, query: str) -> bool:
        """Check if query is safe (only SELECT)."""
        # 移除注释和空白
        cleaned = query.strip().upper()
        
        # 移除可能的注释
        cleaned = cleaned.split("--")[0].split("#")[0].strip()
        
        # 只允许 SELECT
        safe_keywords = {"SELECT", "WITH"}
        for keyword in safe_keywords:
            if cleaned.startswith(keyword):
                return True
        
        return False

            {
                "query": "SELECT id, name, created_at FROM sessions LIMIT 5",
            },

# examples/tools/data_import_tool.py
"""Data import tool for CSV/JSON files."""

import csv
import json
from pathlib import Path
from app.tool.base import ToolDefinition, ToolResult, ToolContext


class DataImportTool(ToolDefinition):
    """Import data from CSV or JSON files."""
    
    @property
    def id(self) -> str:
        return "data_import"
    
    @property
    def description(self) -> str:
        """
        Import data from CSV or JSON files.
        
        Supports:
        - CSV files with headers
        - JSON arrays or objects
        """
        return "从 CSV 或 JSON 文件导入数据"
    
    def parameters_schema(self) -> dict:
        return {
            "type": "object",
            "properties": {
                "file_path": {
                    "type": "string",
                    "description": "文件路径",
                },
                "format": {
                    "type": "string",
                    "enum": ["csv", "json"],
                    "description": "文件格式",
                },
                "parse_headers": {
                    "type": "boolean",
                    "description": "是否解析表头",
                    "default": True,
                },
            },
            "required": ["file_path", "format"],
        }
    
    async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
        """Import data from file."""
        file_path = args["file_path"]
        file_format = args["format"]
        
        try:
            # 验证文件存在
            path = Path(file_path)
            if not path.exists():
                return ToolResult(error=f"文件不存在：{file_path}")
            
            # 读取并解析
            if file_format == "csv":
                data = await self._import_csv(path)
            else:
                data = await self._import_json(path)
            
            # 生成报告
            if isinstance(data, list):
                output = (
                    f"导入成功！\n"
                    f"文件：{file_path}\n"
                    f"格式：{file_format.upper()}\n"
                    f"记录数：{len(data)}\n"
                    f"列名：{list(data[0].keys()) if data else 'N/A'}"
                )
            else:
                output = (
                    f"导入成功！\n"
                    f"文件：{file_path}\n"
                    f"格式：{file_format.upper()}\n"
                    f"键数：{len(data.keys())}"
                )
            
            return ToolResult(
                output=output,
                metadata={"record_count": len(data) if isinstance(data, list) else 1},
            )
        except Exception as e:
            return ToolResult(error=f"导入失败：{str(e)}")
    
    async def _import_csv(self, path: Path) -> list[dict]:
        """Import CSV file."""
        data = []
        
        async with asyncio.to_thread(path.read_text, encoding='utf-8') as content:
            reader = csv.DictReader(content.splitlines())
            for row in reader:
                data.append(row)
        
        return data
    
    async def _import_json(self, path: Path) -> list[dict] | dict:
        """Import JSON file."""
        async with asyncio.to_thread(path.read_text, encoding='utf-8') as content:
            data = json.loads(content)
        
        # 确保返回数组
        if isinstance(data, dict):
            return [data]
        return data


# 使用示例

            user_id="test",
        )

                "file_path": "/tmp/users.csv",
                "format": "csv",
            },
            ctx,
        )

# examples/tools/file_search_tool.py
"""File search tool for finding files."""

import fnmatch
from pathlib import Path
from app.tool.base import ToolDefinition, ToolResult, ToolContext


class FileSearchTool(ToolDefinition):
    """Search for files by pattern."""
    
    @property
    def id(self) -> str:
        return "file_search"
    
    @property
    def description(self) -> str:
        """
        Search for files matching a pattern.
        
        Supports:
        - Wildcards (*, ?)
        - Case-sensitive/insensitive
        - Limit results
        """
        return "按模式搜索文件"
    
    def parameters_schema(self) -> dict:
        return {
            "type": "object",
            "properties": {
                "pattern": {
                    "type": "string",
                    "description": "搜索模式（支持通配符）",
                    "examples": ["*.py", "*.log", "test_*"],
                },
                "path": {
                    "type": "string",
                    "description": "搜索路径（默认工作区）",
                },
                "case_sensitive": {
                    "type": "boolean",
                    "description": "是否区分大小写",
                    "default": False,
                },
                "max_results": {
                    "type": "integer",
                    "description": "最大结果数",
                    "default": 20,
                },
            },
            "required": ["pattern"],
        }
    
    async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
        """Search for files."""
        pattern = args["pattern"]
        search_path = args.get("path", ctx.workspace or ".")
        case_sensitive = args.get("case_sensitive", False)
        max_results = args.get("max_results", 20)
        
        try:
            # 验证路径
            if ctx.workspace:
                resolved_path = self._validate_path(search_path, ctx.workspace)
            else:
                resolved_path = search_path
            
            # 搜索文件
            results = await self._search_files(
                resolved_path,
                pattern,
                case_sensitive,
                max_results,
            )
            
            # 生成报告
            if results:
                output_lines = [
                    f"找到 {len(results)} 个匹配的文件：",
                    "",
                ]
                
                for i, file_path in enumerate(results, 1):
                    size = Path(file_path).stat().st_size
                    output_lines.append(
                        f"{i}. {file_path} ({size:,} bytes)"
                    )
            else:
                output_lines = [
                    f"未找到匹配 '{pattern}' 的文件。"
                ]
            
            return ToolResult(
                output="\n".join(output_lines),
                metadata={"count": len(results)},
            )
        except Exception as e:
            return ToolResult(error=f"搜索失败：{str(e)}")
    
    def _validate_path(self, path: str, workspace: str) -> str:
        """Validate path is within workspace."""
        p = Path(path)
        if not p.is_absolute():
            p = Path(workspace) / p
        
        resolved = p.resolve()
        ws = Path(workspace).resolve()
        
        try:
            resolved.relative_to(ws)
        except ValueError:
            raise ValueError(f"路径超出工作区：{path}")
        
        return str(resolved)
    
    async def _search_files(
        self,
        path: str,
        pattern: str,
        case_sensitive: bool,
        max_results: int,
    ) -> list[str]:
        """Search for files matching pattern."""
        results = []
        
        async with asyncio.to_thread(Path(path).glob, f"**/{pattern}") as files:
            for file_path in files:
                if file_path.is_file():
                    # 处理大小写
                    if case_sensitive:
                        if fnmatch.fnmatch(str(file_path), pattern):
                            results.append(str(file_path))
                    else:
                        if fnmatch.fnmatch(
                            str(file_path).lower(),
                            pattern.lower()
                        ):
                            results.append(str(file_path))
                
                if len(results) >= max_results:
                    break
        
        return results


# 使用示例
if __name__ == "__main__":
    import asyncio
    from app.tool.context import ToolContext
    from unittest.mock import MagicMock
    
    async def main():
        tool = FileSearchTool()
        
        ctx = ToolContext(

            workspace="/tmp",
            session_id="test",
            user_id="test",
        )
        
        result = await tool.execute(
            {
                "pattern": "*.py",
                "max_results": 10,
            },
            ctx,
        )
        
        print(result.output)

# examples/tools/file_analysis_tool.py
"""File analysis tool for examining file contents."""

from pathlib import Path
from app.tool.base import ToolDefinition, ToolResult, ToolContext


class FileAnalysisTool(ToolDefinition):
    """Analyze file structure and content."""
    
    @property
    def id(self) -> str:
        return "file_analysis"
    
    @property
    def description(self) -> str:
        """
        Analyze file structure and content.
        
        Provides:
        - File size and type
        - Line count
        - Encoding detection
        - Code metrics (for code files)
        """
        return "分析文件结构和内容"
    
    def parameters_schema(self) -> dict:
        return {
            "type": "object",
            "properties": {
                "file_path": {
                    "type": "string",
                    "description": "文件路径",
                },
                "metrics": {
                    "type": "array",
                    "items": {
                        "type": "string",
                        "enum": ["size", "lines", "encoding", "code_metrics"],
                    },
                    "description": "要分析的指标",
                    "default": ["size", "lines", "encoding"],
                },
            },
            "required": ["file_path"],
        }
    
    async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
        """Analyze file."""
        file_path = args["file_path"]
        metrics = args.get("metrics", ["size", "lines", "encoding"])
        
        try:
            path = Path(file_path)
            
            if not path.exists():
                return ToolResult(error=f"文件不存在：{file_path}")
            
            # 收集指标
            analysis = {}
            
            if "size" in metrics:
                analysis["size"] = {
                    "bytes": path.stat().st_size,
                    "human": self._format_size(path.stat().st_size),
                }
            
            if "lines" in metrics:
                analysis["lines"] = await self._count_lines(path)
            
            if "encoding" in metrics:
                analysis["encoding"] = await self._detect_encoding(path)
            
            if "code_metrics" in metrics and self._is_code_file(path):
                analysis["code_metrics"] = await self._analyze_code(path)
            
            # 生成报告
            output_lines = [f"文件分析：{file_path}", "=" * 50, ""]
            
            for key, value in analysis.items():
                if isinstance(value, dict):
                    output_lines.append(f"{key.upper()}:")
                    for k, v in value.items():
                        output_lines.append(f"  {k}: {v}")
                else:
                    output_lines.append(f"{key}: {value}")
                output_lines.append("")
            
            return ToolResult(output="\n".join(output_lines))
        except Exception as e:
            return ToolResult(error=f"分析失败：{str(e)}")
    
    def _format_size(self, size: int) -> str:
        """Format file size."""
        for unit in ['B', 'KB', 'MB', 'GB']:
            if size < 1024:
                return f"{size:.1f} {unit}"
            size /= 1024
        return f"{size:.1f} TB"
    
    async def _count_lines(self, path: Path) -> dict:
        """Count lines in file."""
        async with asyncio.to_thread(path.read_text, encoding='utf-8', errors='ignore') as content:
            lines = content.splitlines()
            return {
                "total": len(lines),
                "with_content": len([l for l in lines if l.strip()]),
                "empty": len([l for l in lines if not l.strip()]),
            }
    
    async def _detect_encoding(self, path: Path) -> str:
        """Detect file encoding."""
        import chardet
        
        async with asyncio.to_thread(path.read_bytes) as content:
            result = chardet.detect(content)
            return result["encoding"] or "unknown"
    
    async def _analyze_code(self, path: Path) -> dict:
        """Analyze code metrics."""
        import ast
        
        async with asyncio.to_thread(path.read_text, encoding='utf-8') as content:
            try:
                tree = ast.parse(content)
                
                functions = len([
                    n for n in ast.walk(tree)
                    if isinstance(n, (ast.FunctionDef, ast.AsyncFunctionDef))
                ])
                
                classes = len([
                    n for n in ast.walk(tree)
                    if isinstance(n, ast.ClassDef)
                ])
                
                return {
                    "functions": functions,
                    "classes": classes,
                }
            except SyntaxError:
                return {"error": "不是有效的代码文件"}
    
    def _is_code_file(self, path: Path) -> bool:
        """Check if file is a code file."""
        code_extensions = {
            '.py', '.js', '.ts', '.java', '.c', '.cpp',
            '.rb', '.go', '.rs', '.php', '.swift'
        }
        return path.suffix.lower() in code_extensions


# 使用示例

    from app.tool.context import ToolContext
    from unittest.mock import MagicMock
    
    async def main():
        tool = FileAnalysisTool()

            user_id="test",
        )
        
        result = await tool.execute(
            {
                "file_path": "/tmp/example.py",
                "metrics": ["size", "lines", "code_metrics"],
            },
            ctx,
        )
        
        print(result.output)

# examples/tools/http_request_tool.py
"""HTTP request tool for making web requests."""

import httpx
from app.tool.base import ToolDefinition, ToolResult, ToolContext


class HTTPRequestTool(ToolDefinition):
    """Make HTTP requests to web services."""
    
    def __init__(self, timeout: float = 30.0):
        self.timeout = timeout
    
    @property
    def id(self) -> str:
        return "http_request"
    
    @property
    def description(self) -> str:
        """
        Make HTTP requests to web services.
        
        Supports:
        - GET, POST, PUT, DELETE, PATCH
        - Custom headers and body
        - Query parameters
        """
        return "发送 HTTP 请求"
    
    def parameters_schema(self) -> dict:
        return {
            "type": "object",
            "properties": {
                "url": {
                    "type": "string",
                    "description": "请求 URL",
                },
                "method": {
                    "type": "string",
                    "enum": ["GET", "POST", "PUT", "DELETE", "PATCH"],
                    "description": "HTTP 方法",
                    "default": "GET",
                },
                "headers": {
                    "type": "object",
                    "description": "请求头",
                },
                "params": {
                    "type": "object",
                    "description": "查询参数",
                },
                "body": {
                    "type": "object",
                    "description": "请求体（POST/PUT/PATCH）",
                },
            },
            "required": ["url"],
        }
    
    async def execute(self, args: dict, ctx: ToolContext) -> ToolResult:
        """Execute HTTP request."""
        url = args["url"]
        method = args.get("method", "GET")
        headers = args.get("headers", {})
        params = args.get("params")
        body = args.get("body")
        
        try:
            async with httpx.AsyncClient(timeout=self.timeout) as client:
                response = await client.request(
                    method=method,
                    url=url,
                    params=params,
                    headers=headers,
                    json=body,
                )
                
                # Build response
                output_lines = [
                    f"HTTP {response.status_code} {response.reason_phrase}",
                    f"URL: {response.url}",
                    "",
                    "响应头:",
                ]
                
                for key, value in response.headers.items():
                    output_lines.append(f"  {key}: {value}")
                
                output_lines.extend(["", "响应内容:", response.text])
                
                return ToolResult(
                    output="\n".join(output_lines),
                    metadata={
                        "status_code": response.status_code,
                        "headers": dict(response.headers),
                    },
                )
        except httpx.TimeoutException:
            return ToolResult(error="请求超时")
        except httpx.RequestError as e:
            return ToolResult(error=f"请求失败：{str(e)}")
        except Exception as e:
            return ToolResult(error=f"执行失败：{str(e)}")


# 使用示例
if __name__ == "__main__":
    import asyncio
    from app.tool.context import ToolContext
    from unittest.mock import MagicMock
    
    async def main():
        tool = HTTPRequestTool()
        
        ctx = ToolContext(
            agent=MagicMock(),
            workspace=None,
            session_id="test",
            user_id="test",
        )
        
        # GET request
        result = await tool.execute(
            {
                "url": "https://api.github.com/repos/openyak/openyak",
                "method": "GET",
            },
            ctx,
        )
        
        print(result.output[:500])  # Print first 500 chars
    
    # asyncio.run(main())
